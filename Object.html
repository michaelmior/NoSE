def deep_symbolize_keys
  return each_with_object({}) do |(k, v), memo|
    memo[k.to_sym] = v.deep_symbolize_keys
    memo
  end if is_a? Hash

  return each_with_object([]) do |v, memo|
    memo << v.deep_symbolize_keys
    memo
  end if is_a? Array

  self
end
